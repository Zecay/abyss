<script>
    // 1. Particle Ambiance
    const canvas = document.getElementById('particles');
    const ctx = canvas.getContext('2d');
    let particles = [];
    function init() {
        canvas.width = window.innerWidth; canvas.height = window.innerHeight;
        particles = Array.from({length: 60}, () => ({
            x: Math.random()*canvas.width, y: Math.random()*canvas.height,
            s: Math.random()*2, v: Math.random()*0.5 + 0.1
        }));
    }
    function draw() {
        ctx.clearRect(0,0,canvas.width, canvas.height);
        ctx.fillStyle = "rgba(0, 255, 255, 0.2)";
        particles.forEach(p => {
            ctx.beginPath(); ctx.arc(p.x, p.y, p.s, 0, Math.PI*2); ctx.fill();
            p.y -= p.v; if(p.y < 0) p.y = canvas.height;
        });
        requestAnimationFrame(draw);
    }

    // 2. Scroll Logic (Depth & Reveal)
    window.addEventListener('scroll', () => {
        const scrolled = window.scrollY;
        // Update Depth
        document.getElementById('depth-value').innerText = `${Math.floor(scrolled * 4).toString().padStart(4, '0')}m`;
        // Darken Background
        const opacity = Math.max(0.2, 1 - scrolled / 2000);
        document.getElementById('bg-overlay').style.opacity = opacity;
        
        // Trigger Reveal
        document.querySelectorAll('.reveal').forEach(el => {
            if(el.getBoundingClientRect().top < window.innerHeight * 0.8) el.classList.add('active');
        });
    });

    // 3. Terminal Logs
    const logs = [
        "> DEPTH: 3200m – UNKNOWN ENTITY DETECTED",
        "> CREATURE LOG #042 – STATUS: HOSTILE",
        "> SIGNAL LOST...",
        "> RE-ESTABLISHING UPLINK...",
        "> BIOMASS DETECTED: LARGE",
        "> PRESSURE LEVEL: CRITICAL",
        "> OXYGEN LEVEL: 84%"
    ];
    let logIdx = 0;
    setInterval(() => {
        const container = document.getElementById('log-container');
        const line = document.createElement('div');
        line.className = 'log-line';
        line.innerText = logs[logIdx];
        container.appendChild(line);
        if(container.children.length > 5) container.removeChild(container.firstChild);
        logIdx = (logIdx + 1) % logs.length;
    }, 3000);

    // 4. Countdown Logic
    const launchDate = new Date('2026-02-08T21:00:00').getTime();
    function updateTimer() {
        const now = new Date().getTime();
        const diff = launchDate - now;

        if (diff <= 0) {
            document.getElementById('main-title').innerText = "OPENED";
            document.getElementById('sub-title').innerText = "THE ABYSS IS READY FOR YOU";
            document.getElementById('timer').style.display = 'none';
            document.getElementById('primary-cta').innerText = "PLAY ON ROBLOX";
            return;
        }

        const d = Math.floor(diff / (86400000));
        const h = Math.floor((diff % 86400000) / 3600000);
        const m = Math.floor((diff % 3600000) / 60000);
        const s = Math.floor((diff % 60000) / 1000);

        document.getElementById('days').innerText = d.toString().padStart(2, '0');
        document.getElementById('hours').innerText = h.toString().padStart(2, '0');
        document.getElementById('mins').innerText = m.toString().padStart(2, '0');
        document.getElementById('secs').innerText = s.toString().padStart(2, '0');
    }

    window.addEventListener('resize', init);
    init(); draw();
    setInterval(updateTimer, 1000); updateTimer();
</script>
